```nginx
# Process Configuration
user nginx;
# - Specifies which user runs the NGINX worker processes
# - Security best practice: use dedicated non-root user
# - Reduces potential damage if compromised

worker_processes auto;
# - Number of worker processes NGINX will spawn
# - 'auto' sets it to match CPU core count
# - Optimizes performance based on available resources

# Connection Configuration
events {
    worker_connections 1024;
    # - Maximum connections each worker process can handle
    # - Total concurrent connections = worker_processes Ã— worker_connections
    # - 1024 is a good default for most scenarios
}

# HTTP Server Configuration
http {
    include /etc/nginx/mime.types;
    # - Includes MIME type definitions
    # - Maps file extensions to content types
    # - Ensures proper content-type headers in responses

    server {
        # SSL/TLS Configuration
        listen 443 ssl;
        listen [::]:443 ssl;
        # - Listen on port 443 (HTTPS) with SSL enabled
        # - Second line handles IPv6 connections
        
        server_name localhost;
        # - Defines server name for virtual hosting
        # - Should match your domain (e.g., login.42.fr)
        
        # SSL Certificate Configuration
        ssl_certificate /etc/nginx/ssl/nginx.crt;
        ssl_certificate_key /etc/nginx/ssl/nginx.key;
        # - Paths to SSL certificate and private key
        # - Generated by Dockerfile during build
        
        ssl_protocols TLSv1.2 TLSv1.3;
        # - Specifies allowed TLS versions
        # - Project requirement: only TLSv1.2/1.3
        # - Enhances security by excluding older versions
        
        # WordPress Root Configuration
        root /var/www/html;
        # - Document root directory
        # - Where WordPress files are stored
        # - Mapped to WordPress volume
        
        index index.php;
        # - Default index file
        # - WordPress uses index.php as main entry point
        
        # URL Processing
        location / {
            try_files $uri $uri/ /index.php?$args;
            # - Handles WordPress pretty URLs
            # - Tries to match:
            #   1. Exact URI
            #   2. URI with trailing slash
            #   3. Passes to index.php with query string
        }
        
        # PHP Processing
        location ~ \.php$ {
            fastcgi_pass wordpress:9000;
            # - Forwards PHP requests to WordPress container
            # - 'wordpress' is the service name in docker-compose
            # - Port 9000 is PHP-FPM's default port
            
            include fastcgi_params;
            # - Includes standard FastCGI parameters
            # - Required for proper PHP processing
            
            fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
            # - Sets full path to PHP script
            # - Combines document root with script name
            # - Required for PHP-FPM to locate scripts
        }
    }
}
```
